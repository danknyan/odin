# Preprocess text (username and link placeholders)
def preprocess(text:str,remove_nonlexical=False,return_list=True,model_size=514):
    # skip empty
    if not text:
        return ""
    #remove newline and split to list
    if isinstance(text,str):
        # skip blanks and non-Latin
        if not (text[-1].isascii() or text[0].isascii()):
            return ""
        text = text.replace("\n","") # remove new line
        # tokenize or just split
        if remove_nonlexical:
            text_as_list = word_tokenize(text)
        else:
            text_as_list = text.split(" ")
    #empty list
    new_text = []
    # for each word in a text
    for t in text_as_list:
        # skip if blank or non-latin
        if (not t or not t[0].isascii()):
            continue
        # anonymization
        t = '@user' if t.startswith('@') and len(t) > 1 else t
        t = 'http' if t.startswith('http') else t
        # lexical processing
        if remove_nonlexical:
            t = remove_punctuation(t) # remove punctuation
            #remove stopwords and blanks
            if (not t or (t in stop_words or t in "nt") or t[-1].isnumeric()):
                continue
        new_text.append(t)
    #limit by model buffer size
    if return_list:
        return " ".join(new_text)[:model_size]
    return new_text

# sentiment scores from from str
def process_comment_from_df(raw_text:str) -> tuple[float,float,float]:
    # empty vars
    sentiments=[]
    # preprocess text
    text = preprocess(raw_text)
    encoded_input = tokenizer(text, return_tensors='pt')
    # model output
    try:
        output = model(**encoded_input)
        scores = softmax(output[0][0].detach().numpy())
    
    except Exception as e:
        print("raw text:",raw_text)
        print("processed text:",text)
        print(str(e))
        #print(traceback.format_exc())
        scores = np.zeros(3)
    
    ranking = np.argsort(scores)[::-1]
    # extract labels and scores
    for i in range(scores.shape[0]):
        s = scores[ranking[i]]
        sentiments.append(np.round(float(s),3))
        # expected output: "Covid cases are increasing fast!"
        # 1) Negative 0.7236
        # 2) Neutral 0.2287
        # 3) Positive 0.0477
    return sentiments

def avg_sentiment(df):
    sentiment_type = ["negative","positive"]
    average_sentiment = {}
    for sentiment in sentiment_type:
        avg = df[sentiment].mean()
        std = df[sentiment].std()
        average_sentiment[sentiment] = (max(avg-std,0),avg+std)
    return average_sentiment

def get_comments(video_tag="tybOi4hjZFQ",verbose=True,api_key=api_key) -> tuple[pd.DataFrame, str]:
    youtube = build('youtube', 'v3', developerKey=api_key)

    # get title if needed
    video_request=youtube.videos().list( 
        part='snippet', 
        id=video_tag
        ) 
    video_response = video_request.execute()
    vid_title = video_response['items'][0]['snippet']['title']

    # get comments
    request = youtube.commentThreads().list(
        part="snippet,replies",
        videoId=video_tag,
        textFormat="plainText"
    )
    # init empty df
    df = pd.DataFrame(columns=['comment', 'replies', 'date', 'user_name'])


    while request:
        replies = []
        comments = []
        dates = []
        user_names = []
       

        try:
            response = request.execute()
            #counter for verbosity
            cycle_count=0
            
            for item in response['items']:
                # Extracting comments
                comment = item['snippet']['topLevelComment']['snippet']['textDisplay']
                comments.append(comment.lower())

                user_name = item['snippet']['topLevelComment']['snippet']['authorDisplayName']
                user_names.append(user_name)

                date = item['snippet']['topLevelComment']['snippet']['publishedAt']
                dates.append(date)

                # counting number of reply of comment
                replycount = item['snippet']['totalReplyCount']

                # if reply is there
                if replycount > 0:
                    # append empty list to replies
                    replies.append([])
                    # iterate through all reply
                    for reply in item['replies']['comments']:
                        # Extract reply
                        reply = reply['snippet']['textDisplay']
                        # append reply to last element of replies
                        replies[-1].append(reply)
                else:
                    replies.append([])
            # create new dataframe
            df2 = pd.DataFrame.from_dict({"comment": comments, "replies": replies, "user_name": user_names, "date": dates},orient="index").transpose()
            # update existing df 
            df = pd.concat([df, df2], ignore_index=True)
            # save df just in case
            df.to_json(f"./data/user_comments_{video_tag}.json")
            #get next page
            request = youtube.commentThreads().list_next(request, response)
            #report progress
            if (cycle_count % 50 == 0) and (verbose == True):
                print(f"Iterating. {df.shape[0]} comments detected thus far.")
            cycle_count +=1
        except Exception as e:
            print(f"Reached suspected end-of-page. Exporting to {video_tag}_user_comments.json .")
            df.to_json(f"./data/user_comments_{video_tag}.json")
            print(str(e))
            print(traceback.format_exc())
    return df,vid_title

# extract sentiments
def sentiment_analysis(df:pd.DataFrame) -> pd.DataFrame:
    senti = df["comment"].map(lambda x: process_comment_from_df(preprocess(x)))
    negative = list(map(lambda x: x[2],senti))
    neutral = list(map(lambda x: x[1],senti))
    positive = list(map(lambda x: x[0],senti))
    senti_df = pd.DataFrame.from_dict({"negative":negative,"neutral":neutral,"positive":positive})
    return pd.concat([senti_df,df],axis=1)

def build_dictionary(df:pd.DataFrame,return_ngrams=False) -> Counter:
    unigrams_freq = Counter()
    bigrams_freq = Counter()
    bigrams = []
    # TODO: fix bigram generator
    # get tokens
    tokens = df["comment"].map(lambda x: word_tokenize(preprocess(x,remove_nonlexical=True)))
    # get ngrams
    for comment in tokens:
        comment_bigrams = []
        for i in range(len(comment)-1):
            comment_bigrams.append(" ".join(comment[i:i+2]))
        bigrams.append(comment_bigrams)
    # update dictionary
    list(map(lambda x:unigrams_freq.update(x),tokens))
    # update bigrams counter
    list(map(lambda x:bigrams_freq.update(x),bigrams))

    # add columns
    df["unigrams"]=tokens
    df["bigrams"]=bigrams
    if return_ngrams:
        return unigrams_freq,bigrams_freq

# general ngram extractor, default bigram
def extract_ngrams(text:str,ngram_count:int=2) -> Counter:
    ngram_counter = Counter()
    text_tokenized = preprocess(text,remove_nonlexical=True,return_list=False)
    ngram_list = []
    for i in range(len(text_tokenized)-ngram_count+1):
        token_list = text_tokenized[i:i+ngram_count]
        ngram = " ".join(token_list)
        ngram_list.append(ngram)
    ngram_counter.update(ngram_list)    
    return ngram_counter

# get ngram polarity
def get_ngram_polarity(df,sentiment_thresholds={"negative":(.05,0.10),"positive":(0.75,0.99)}) -> dict:
    # setup vars
    pos_tokens = Counter()
    neg_tokens = Counter()
    for _,row in df.iterrows():
        # check polarity, make entry
        if row["negative"] >= max(sentiment_thresholds["negative"]) and \
            row["positive"] < min(sentiment_thresholds["positive"]):
            neg_tokens.update(row["unigrams"])
            neg_tokens.update(row["bigrams"])
        elif row["positive"] >= min(sentiment_thresholds["positive"]):
            pos_tokens.update(row["unigrams"])
            pos_tokens.update(row["bigrams"])
    return pos_tokens,neg_tokens
